  try {
    // ... database operations ...

  } catch (error) {
    logger.ERROR('Failed to add resident', { error: error.message });
    next(error);
  }

  // Check server, database, memory

// 5. Admin backup route
app.post('/admin/backup', requireAdmin, (req, res) => {
  const result = backupDatabase('database.db');
  res.json(result);
});

// 6. 404 handler (BEFORE error middleware)
app.use((req, res) => {
  res.status(404).render('error-404', { path: req.path });
});

// 7. Error middleware (LAST)
app.use((err, req, res, next) => {
  logger.ERROR('Request error', {
    error: err.message,
    stack: err.stack,
    path: req.path
  });
  
  res.status(500).render('error', {
    message: 'May problema sa system',
    debug: process.env.NODE_ENV === 'development' ? err.stack : null
  });
});

// 8. Automatic backups (2:00 AM daily)
cron.schedule('0 2 * * *', () => {
  logger.INFO('Running automatic backup');
  backupDatabase('database.db');
  cleanOldBackups('backups', 7);
});

// 9. Start server
const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  logger.INFO('Server started', { port: PORT, env: process.env.NODE_ENV });
});
